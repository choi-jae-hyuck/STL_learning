#include <iostream>
#include <deque>
#include <vector>
#include <iterator>
#include <fstream>
#include "String.h"
#include "save.h"
using namespace std;

// array<T>     [i] 연속적
// vector<T>    [i] 연속적
// deque<T>     [i] 파편화됨 속도를 희생한대신 기능부여
// list<T>

// 소스파일을 뒤에서 부터 출력하라.

int main()
{
	
	ifstream in("소스.cpp");
	
	deque<char> d;

	copy(istreambuf_iterator<char>(in), istreambuf_iterator<char>(), front_inserter(d));
	copy(d.begin(), d.end(), ostreambuf_iterator<char>(cout));
	

	in.close();

	save("소스.cpp");
}
