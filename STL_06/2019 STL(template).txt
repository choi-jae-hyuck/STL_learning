#include <iostream>
#include <vector>
#include <algorithm>
#include<iterator>
#include "String.h"
#include "save.h"
using namespace std;

template <typename InIter,typename OutIter>
void myCopy(InIter beg, InIter end, OutIter des);

int main()
{
	vector<String> v{ 1,2,3,4,5 };
	vector<String> w;
	myCopy(v.begin(), v.end(), back_inserter(w));//iterator adaptor 
	
	save("소스.cpp");
}

template <class InIter, class OutIter>
void myCopy(InIter beg, InIter end, OutIter des)
{
	while (beg != end)
		*des++ = *beg++;
}
