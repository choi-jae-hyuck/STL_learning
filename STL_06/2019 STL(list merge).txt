#include <iostream>
#include <list>
#include "String.h"
#include "save.h"
using namespace std;

// list에 String 객체 10,20,30을 추가하라
// 길이가 20의 배수인 객체를 지워라.
// 화면에 list의 원소 전체를 출력하라

int main()
{
	list<String> v{ 3,1,4,7,8,9,2,1,4,56 };   //vector
	list<String> v2{ 2,9,1,1,6,4,5,8,7,9,3,3 };
	
	//merge는 정렬한 list에 사용하는것
	// v.sort( < )
	v.sort();
	v2.sort();
	v.merge(v2);

	v.unique();
	cout << "------------" << endl;
	for (const String& d : v)
		cout << d << endl;
	cout << "------------" << endl;
	save("소스.cpp");
}
