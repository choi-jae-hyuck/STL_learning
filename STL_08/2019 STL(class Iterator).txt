#include <iostream>
#include <vector>
#include <algorithm>
#include <numeric>
#include "save.h"
#include "String.h"
using namespace std;
using namespace chrono;


template <class InIter, class OutIter>
void myCopy(InIter b, InIter e, OutIter d)
{
	while (b != e) {
		*d = *b;
		++d;
		++b;
	}
}


int main()
{
	String s(100);
	cout << s << endl;

	for (char c : s)
		cout << c;
	cout<<endl;

	for (auto i = s.begin(); i != s.end(); ++i)
		cout << *i;
	cout << endl;

	for (auto i = s.rbegin(); i != s.rend(); ++i)
		cout << *i;
	cout << endl;

	//sort(s.begin(), s.end());
	cout << s << endl;
